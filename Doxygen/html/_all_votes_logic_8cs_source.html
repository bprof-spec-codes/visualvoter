145  public bool IsVoteWon(int voteID)
146  {
147  //According to the .pdf, a vote is won, when
148  // - 1.) It has the most yes votes in it's voting group,
149  //AND
150  // - 2.) 2/3 of the voters, who submitted a vote to this, or any other candidate in the same voting group, voted 'Yes' to this candidate.
151 
152 
153 
154  var thisVote = this.GetOneVote(voteID);
155 
156  if (string.IsNullOrWhiteSpace(thisVote.voteGroup) || thisVote.voteGroup.ToLower() == "string")
157  {
158  var totalVotes = thisVote.YesVotes + thisVote.NoVotes + thisVote.AbsentionVotes;
159  var twoThird = totalVotes / 3.0 * 2.0;
160  if (thisVote.YesVotes >= twoThird)
161  {
162  return true;
163  }
164  return false;
165  }
166 
167  bool mostYesVoted_Condition = false; // 1.)
168  bool twoThird_Condition = false; // 2.)
169 
170  var otherVotesInSameGroup = getVotesFromGroup(thisVote.voteGroup);
171  int maxYesVotesInGroup = -1;
172  foreach (var item in otherVotesInSameGroup)
173  {
174  if (item.YesVotes > maxYesVotesInGroup)
175  {
176  maxYesVotesInGroup = item.YesVotes;
177  }
178  }
179  if (thisVote.YesVotes >= maxYesVotesInGroup)
180  {
181  mostYesVoted_Condition = true;
182  }
183 
184  int participantNumber = this.numberOfGroupParticipants(thisVote.voteGroup);
185  double twoThirds = participantNumber / 3.0 * 2.0;
186  if (thisVote.YesVotes >= twoThirds)
187  {
188  twoThird_Condition = true;
189  }
190  return mostYesVoted_Condition && twoThird_Condition;
191  }
